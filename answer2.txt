def handan(num)
    cont = 0
    for i in 2..num
        if num%i == 0 
            cont += 1 
    end
    end
    
    if cont < 2 then return 1
    else return 0
    end
end

for num in 2..100
    tmp = handan(num)
    if tmp == 1 then puts num
    end
end
