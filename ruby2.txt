def isLeapYear(year) 
  return false if year < 4

  year % 400 == 0 || (year % 100 != 0 && year % 4 == 0) 
end

def countLeapYear(fromYear, toYear) 
  (fromYear..toYear).count{|i| isLeapYear i}
end
