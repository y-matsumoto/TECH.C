# 素数
def get_is_prime(b)
    (2..Math.sqrt(b)).each do |c|
        if b % c == 0
            return false
        end
    end
    return true
end

(2..100).each do |a|
    if get_is_prime(a) 
        p a
    end
end
    

# 閏年
def is_uru(year)
    if year%400 == 0
        return true
    end
    if year%100 == 0
        return false
    end
    if year%4 == 0
        return true
    end
end

(0..2014).each do |year|
    if is_uru(year)
        p year
    end
end
