def generate_prime_numbers(max)
  (2..max).inject([]) do |list, n|
    append_if_prime(list, n)
  end
end

def append_if_prime(known_primes, n)
  known_primes.each {|prime| return known_primes if n % prime == 0 } 
  known_primes << n
end
